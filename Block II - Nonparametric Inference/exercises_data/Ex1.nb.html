Calcolo il valore del test statistics per le permutazioni

seed=12782
B= 10000
n= length(x[,1])
n1= length(g1[,1])
n2= length(g2[,1])

T2 = numeric(B)
set.seed(seed)
for(perm in 1:B){
  # Random permutation of indexes
  # When we apply permutations in a multivariate case, we keep the units together i.e., we only permute the rows of the data matrix
  t_pooled = rbind(g1,g2)
  permutation = sample(n)
  t_perm = t_pooled[permutation,]
  t1_perm = t_perm[1:n1,]
  t2_perm = t_perm[(n1+1):n,]
  
  # Evaluation of the test statistic on permuted data
  t1.med_perm = depthMedian(t1_perm,depth_params = list(method='Mahalanobis'))
  t2.med_perm = depthMedian(t2_perm,depth_params = list(method='Mahalanobis'))
  T2[perm]  = max(abs(t1.med_perm-t2.med_perm))
}
